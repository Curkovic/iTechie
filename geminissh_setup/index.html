if [[ "$SSH_CONNECTION" ]] && [[ "$USER" == "bot" ]]; then
    GREETING=$(~/gemini.sh "Generate a single short, friendly SSH login greeting, be funny. No lists, no options, no explanations.")
    echo "$GREETING"
fi

CPU=$(top -bn1 | grep "Cpu(s)" | awk '{print $2 + $4 "%"}')
RAM=$(free -h | awk '/^Mem/ {print $3 "/" $2}')
DISK=$(df -h / | awk 'NR==2 {print $3 "/" $2 " (" $5 " used)"}')
UPTIME=$(uptime -p)
DEVICE="$USER@$(hostname -I | awk '{print $1}')"

# System health check
~/gemini.sh "Device: $DEVICE CPU: $CPU RAM: $RAM Disk: $DISK Uptime: $UPTIME Only respond if something is abnormal or needs attention. If everything is fine, just say something like All systems normal. and add super short comment about system. Keep it very short."
